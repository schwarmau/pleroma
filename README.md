```
git init

git checkout origin/master
cd pleroma

git fetch
git checkout origin/stable
cd ../pleroma-fe
git init
git remote add origin https://git.pleroma.social/pleroma/pleroma-fe.git
git fetch
git checkout origin/master
cd ..
```
